massive-automator/
├─ pyproject.toml
├─ README.md
├─ .gitignore
├─ Makefile
├─ configs/
│  ├─ macro_map.yaml           # Macro → CC map + destinations you MIDI-learned in Massive
│  └─ generators.yaml          # Batch recipes (e.g., 50 basses, 30 leads)
├─ presets/

├─ scripts/
│  └─ setup_iac_macos.sh       # Creates/ensures IAC virtual MIDI bus (macOS)
├─ src/
│  └─ massive_automator/
│     ├─ __init__.py
│     ├─ schema.py             # Dataclasses defining a patch (OSC/Filter/Env/LFO/FX/Global)
│     ├─ randomizers.py        # Musically-constrained generators (lead/bass/pad/pluck)
│     ├─ validate.py           # JSON schema / dataclass validation
│     ├─ midi.py               # Port discovery + CC sender (mido/rtmidi)
│     ├─ macros.py             # Map patch → 8 macro values (0..127) using macro_map.yaml
│     ├─ io_json.py            # Load/save patch JSON; batch I/O utilities
│     └─ cli.py                # Typer CLI entrypoint
└─ tests/
   ├─ test_schema.py
   ├─ test_macros.py
   └─ test_midi.py
